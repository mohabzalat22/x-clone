Table users {
  id int [PK]
  name varchar
  email varchar
  password varchar
  profile_picture varchar
  background_picture varchar
  bio varchar
}

Table tweets {
  id int [Pk]
  user_id int [ref: > users.id]
  data varchar
  source varchar
}

Table reactions {
  id int [Pk]
  user_id int [ref: > users.id]
  tweet_id int [ref: > tweets.id]
  heart bool
  repost bool
  viewed bool
}

Table comments {
  id int [Pk]
  user_id int [ref: > users.id]
  tweet_id int [ref: > tweets.id]
  comment varchar
}

Table friends {
  id int [PK] 
  follower_id int [ref: > users.id]
  following_id int
}
